29. wait(NULL)很关键，让子进程先运行结束，否则则是父进程先，而导致检测失败

​```c++
#include<iostream>
#include<cstdlib>
#include <unistd.h>
#include<sys/wait.h>
using namespace std;
int pwncollege() {
        return 0;
}

int main() {
        int pid = fork();

        if(pid == -1) {
                cout << "fork error";
        }

        else if (pid == 0) {
                int test;
                if(execl("/challenge/embryoio_level29", NULL) == -1) {
                        cout << "execl error";
                        exit(-1);
                }
        }
        else {
                wait(NULL);
                cout<<"crazy";
                exit(0);
        }
        return 0;
}
​```
