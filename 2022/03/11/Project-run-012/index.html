78. 不是用管道，而是输入重定向，并且要用指定工作目录

79. argv里用cd不行，cd并不是bin目录中的玩意，而是系统自带的

cwd是设置workspace，executable是指定argv[0]的可执行文件所在目录（而不是命令行的意思）

​```python
from pwn import *

p=open("ifozkt", "w")
sh=process(["/challenge/embryoio_level79"], stdin=p.fileno(),cwd="/tmp/doektz")
sh.interactive()
​```

#include <unistd.h>
#include <cstring>
#include <stdlib.h>
#include <stdio.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <iostream>

#define NUM 300

using namespace std;

int pwncollege();

int main() {
    pid_t fd;
    
    char * cmd[] = {"/challenge/embryoio_level80", NULL};
    char * argv[NUM] = {"/challenge/embryoio_level80", NULL};
    char * envp[] = {NULL};
    for(int i = 1; i < NUM -1; i++) {
        argv[i] = "bzudqxryyw";
    }
    argv[NUM - 1] = NULL;
    
    if((fd=fork())==0){
        execve(cmd[0], argv, envp);
    }
    int s1;
    waitpid(fd, &s1, 0);
    return 0;
}

    return 0;
}
